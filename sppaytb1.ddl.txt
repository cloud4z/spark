DROP TABLESPACE FEATURE;
COMMIT;
CREATE  TABLESPACE FEATURE
 USING STOGROUP SYSDEFLT
   PRIQTY -1
   SECQTY -1
 SEGSIZE 32
 BUFFERPOOL BP0
 CCSID EBCDIC
 LOCKMAX SYSTEM
 LOCKSIZE PAGE
 MAXROWS 255;

	
CREATE TABLE CLIENT_JOIN  (
 "CUSTOMER_ID" VARCHAR(20) NOT NULL,
 "GENDER" VARCHAR(1),
 "AGE_YEARS" DECIMAL(5,2),
 "HIGHEST_EDU" VARCHAR(1),
 "ANNUAL_INVESTMENT_REV" BIGINT,
 "ANNUAL_INCOME" BIGINT,
 "TOTAL_TXNS" SMALLINT,
 "AVG_DAILY_TXNS" DECIMAL(4,2),
 "TOTAL_TXN_AMOUNT" DECIMAL(10,2),
 "AVG_TXN_AMOUNT" DECIMAL(10,2),
 "ACTIVITY_LEVEL" VARCHAR(1),
 "CHURN" VARCHAR(1),
 PRIMARY KEY(CUSTOMER_ID)
 )
 AUDIT NONE
 DATA CAPTURE NONE
 CCSID EBCDIC  IN FEATURE;

 CREATE UNIQUE INDEX "IDX_CLIENT_JOIN"
  ON CLIENT_JOIN
  ("CUSTOMER_ID" ASC)
  CLUSTER
  USING STOGROUP SYSDEFLT
    PRIQTY -1
    SECQTY -1
